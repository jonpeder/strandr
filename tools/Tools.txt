# Packages
install_packages("devtools")
install_packages("roxygen2)

# Import strand data from shape file ('strandsystemet 4.0' provided by Anders EndrestÃ¸l)
strand <- readOGR(dsn = ".", layer = "strandsystemet 4.0")

# if you are on a windows pc and have problems with encoding ASCII characters, convert from  utf-8 to latin1 encoding
for (i in c(1,2,4,5,6,7) {
  strand[[i]] <- iconv(strand[[i]], from = "utf8", to = "latin1")
}

# Store data in 'sysdata.rda' in R package
use_data(strand, kommuner, internal = TRUE, overwrite = TRUE)

# Read gml file from GeoNorge
sf::st_layers("../../../../Downloads/Basisdata_0000_Norge_4258_Kommuner_GML.gml") # List layers
g = sf::st_read("../../../../Downloads/Basisdata_0000_Norge_4258_Kommuner_GML.gml", layer = "Kommune") # Read specified layer
kommuner = as(g, "Spatial") # Convert to SpatialPolygonsDataFrame
